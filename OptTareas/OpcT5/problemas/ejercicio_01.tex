\begin{problema}
    \(\gamma\) denota la circunferencia unitaria. Evalúa las integrales

    \begin{enumerate}
        \item \(\int_{\gamma}^{}\tfrac{1}{z}\odif{z}\).
        \item \(\int_{\gamma}^{}\tfrac{1}{\abs{z}}\odif{z}\).
        \item \(\int_{\gamma}^{}\tfrac{1}{z}\abs{\odif{z}}\).
        \item \(\int_{\gamma}^{}\abs`\big{\tfrac{1}{z}\odif{z}}\).
        \item \(\int_{\gamma}^{}z \mathrm{e}^{z^{2}}\odif{z}\).
    \end{enumerate}
\end{problema}
