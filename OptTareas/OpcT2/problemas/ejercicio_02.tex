\begin{problema}
	Pruebe que:

	\begin{equation}
		1 + \cos\bigl(\theta\bigr) + \cdots + \cos\bigl(n\theta\bigr) = \dfrac{1}{2} +
		\dfrac{\sin\Bigl[\bigl(n + \tfrac{1}{2}\bigr)\theta\Bigr]}{2
			\sin\bigl(\tfrac{\theta}{2}\bigr)},
	\end{equation}

	donde \(\theta\) no es un múltiplo par de \(\pi\).
\end{problema}
