\begin{problema}
    sean \(\Omega = \{z\in \mathbb{C} \mid \Re(z) > 1\}\) y \(f \colon \Omega
    \subseteq \mathbb{C} \to \mathbb{C}\) analítica en \(\Omega\). Si \(f(z) = u(z) +
    iv(z)\) y

    \begin{equation*}
        \pdv{u(z)}{x} + \pdv{v(z)}{y} = 0
    \end{equation*}

    para toda \(z\in \omega\), pruebe que existen \(c\in \mathbb{R}\) y
    \(d\in \mathbb{C}\)
    tales que \(f(z) = icz + d\) para toda \(z\in \Omega\).
\end{problema}
