\begin{problema}
    Sea \(\gamma(t) = r \mathrm{e}^{it}, t\in [0, 2\pi]\). Calcule la integral

    \begin{equation*}
        \int_{\gamma}^{}\dfrac{\abs{\odif{z}}}{\abs{z - a}^{2}}
    \end{equation*}

    en donde \(\abs{a} \neq r\). \emph{Sugerencia: recuerde que \(z \overline{z} =
    \abs{z}^{2}\)}.
\end{problema}
