\begin{problema}
    Sea \(\gamma(t) = r \mathrm{e}^{it}, t\in [0, 2\pi]\). Si \(f\) es continua sobre
    \(\gamma\), pruebe que

    \begin{equation}
        \int_{\gamma}^{}f(z)\abs{\odif{z}} = ir \int_{\gamma}^{}\dfrac{f(z)}{z}\odif{z}.
    \end{equation}
\end{problema}
