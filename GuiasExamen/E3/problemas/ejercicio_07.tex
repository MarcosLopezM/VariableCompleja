\begin{problema}
    Sea \(\Omega \subseteq \mathbb{C}\) una región. Definimos \(\tilde{\Omega} =
    \{z\in \mathbb{C} \mid \overline{z}\in \mathbb{C}\}\).

    \begin{enumerate}
        \item ¿\(\tilde{\Omega}\) es una región? Identifique geométricamente a
            \(\tilde{\Omega}\).
        \item Dada \(f \colon \Omega \subseteq \mathbb{C} \to \mathbb{C}\),
            definimos \(g
            \colon \tilde{\Omega} \subseteq \mathbb{C} \to \mathbb{C}\) como \(g(z) =
            \overline{f(z)}\). Pruebe que \(f\) es analítica en \(\Omega\) si
            y solo si \(g\)
            es analítica en \(\tilde{\Omega}\).
    \end{enumerate}
\end{problema}
