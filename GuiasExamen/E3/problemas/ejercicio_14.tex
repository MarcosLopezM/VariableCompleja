\begin{problema}
    Si \(f\) está definida y es continua en la semi-banda horizontal \([x_{0}, \infty)
    \mul [0, h]\) y

    \begin{equation*}
        \lim_{x \to \infty} f(x + iy) = A
    \end{equation*}

    para cada \(y\in [0, h]\), pruebe que

    \begin{equation*}
        \lim_{x \to \infty} \int_{\beta_{x}}^{}f(z)\odif{z} = iAh,
    \end{equation*}

    donde \(\beta_{x}(t) = x + it\) con \(t\in [0, h]\).
\end{problema}
