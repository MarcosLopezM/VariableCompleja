\begin{problema}
    Pruebe las siguientes afirmaciones

    \begin{enumerate}
        \item Si \(f\) está definida y es continua en una vecindad del 0.

            \begin{equation*}
                \lim_{r \to 0} \int_{0}^{2\pi}f(r \mathrm{e}^{it})\odif{t} = 2\pi f(0).
            \end{equation*}

        \item Si \(f\) está definida y es continua en una vecindad del punto
            \(a\), pruebe que

            \begin{equation*}
                \lim_{r \to 0} \int_{\gamma_{r}}^{}\dfrac{f(z)}{z -
                a}\odif{z} = 2\pi if(a).
            \end{equation*}

            donde \(\gamma_{r}(t) = a + r \mathrm{e}^{it}\) con \(t\in [0, 2\pi]\).
    \end{enumerate}
\end{problema}
