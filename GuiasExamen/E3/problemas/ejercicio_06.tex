\begin{problema}
    Sea \(f \colon \mathbb{C} \to \mathbb{C}\) tal que \(f(z + w) = f(z)f(w)\) para
    toda \(z,w\in \mathbb{C}\). Pruebe que

    \begin{enumerate}
        \item \(f(0) = 0\) o \(f(0) = 1\).
        \item Si \(f(0) = 0\), entonces \(f\) es la función constante 0.
        \item Si \(f(0) = 1\) y \(f\) es derivable en \(z = 0\), entonces \(f\) es
            analítica en \(\mathbb{C}\).
        \item Existe \(c\in \mathbb{C}\) tal que \(f(z) = \mathrm{e}^{cz}\) para toda
            \(z\in \mathbb{C}\).
    \end{enumerate}
\end{problema}
