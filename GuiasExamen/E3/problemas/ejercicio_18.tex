\begin{problema}
    Sea \(\Omega = \mathbb{C}\setminus [0, 1]\) y \(f \colon \Omega \subseteq
    \mathbb{C} \to \mathbb{C}\) definida como \(f(z) = \tfrac{1}{z(z - 1)}\). Pruebe que

    \begin{equation*}
        \int_{\gamma}^{}f(z)\odif{z} = 0
    \end{equation*}

    para toda curva cerrada suave por pedazos \(\gamma \subseteq \Omega\).
\end{problema}
