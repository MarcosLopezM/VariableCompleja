\begin{problema}
    Sea

    \begin{equation*}
        f(z) =
        \begin{dcases}
            \dfrac{z^{5}}{\abs{z}^{4}} & \text{si}\; z \neq 0,\\
            0 & \text{si}\; z = 0.
        \end{dcases}
    \end{equation*}

    \begin{enumerate}
        \item Pruebe que si \(u = \Re(f(z))\) y \(v = \Im(f(z))\), entonces
            \(u\) y \(v\)
            satisfacen las ecuaciones de Cauchy-Riemann en \(z = 0\).
        \item ¿\(f\) es complejo derivable en \(z = 0\)?
        \item ¿\(f\) es real-diferenciable en \(z = 0\)?
    \end{enumerate}
\end{problema}
