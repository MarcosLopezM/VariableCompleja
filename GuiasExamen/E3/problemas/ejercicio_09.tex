\begin{problema}
    Si \(\gamma(t) = \mathrm{e}^{it}, t\in [0, 2\pi]\) y \( f\) es una función
    continua sobre la circunferencia unitaria, pruebe que

    \begin{equation*}
        \overline{\int_{\gamma}^{}f(z)\odif{z}} = -
        \int_{\gamma}^{}\overline{f(z)}\overline{z}^{2}\odif{z}.
    \end{equation*}
\end{problema}
