\begin{problema}
    Si \(f\) está definida y es continua para \(\abs{z} > R > 0\) y

    \begin{equation*}
        \lim_{z \to \infty} \dfrac{f(z)}{z} = A,
    \end{equation*}

    pruebe que

    \begin{equation*}
        \lim_{r \to \infty} \int_{\Gamma_{r}}^{}\dfrac{f(z)}{z^{2}}\odif{z} = iA\alpha,
    \end{equation*}

    donde \(\Gamma_{r}(t) = r \mathrm{e}^{it}\) con \( t\in [-\alpha, 0]\) y
    \(0 < \alpha
    \leq 2\pi\).
\end{problema}
