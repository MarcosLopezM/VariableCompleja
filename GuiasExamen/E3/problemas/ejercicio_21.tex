\begin{problema}
    Sean \(f \colon \Omega \subseteq \mathbb{C} \to \mathbb{C}\) tal que
    \(f^{\prime}\) es continua en \(\Omega\), \(z_{0}\in \Omega\) y \(\gamma \colon
    [a, b] \subseteq \mathbb{R} \to \Omega \subseteq \mathbb{C}\) una curva suave por
    pedazos, tales que \(f(\gamma(t)) \neq f(z_{0})\) para toda \(t\in [a, b]\). Si
    \(\tilde{\gamma} = f \circ \gamma\), pruebe que

    \begin{equation*}
        n(\tilde{\gamma}, f(z_{0})) = \dfrac{1}{2\pi
        i}\int_{\gamma}^{}\dfrac{f^{\prime}(z)}{f(z) - f(z_{0})}\odif{z}.
    \end{equation*}
\end{problema}
